**4. Custom service worker** — Create packages/web/src/sw.ts:
```typescript
import { precacheAndRoute, cleanupOutdatedCaches } from 'workbox-precaching';

declare let self: ServiceWorkerGlobalScope;

// Workbox precaching (vite-plugin-pwa injects __WB_MANIFEST)
cleanupOutdatedCaches();
precacheAndRoute(self.__WB_MANIFEST);

// Push notification handler
self.addEventListener('push', (event) => {
  const data = event.data?.json() ?? {};
  event.waitUntil(
    self.registration.showNotification(data.title ?? 'Age No Concern', {
      body: data.body ?? '',
      icon: '/icons/icon-192x192.png',
      badge: '/icons/icon-192x192.png',
      data: { url: data.url ?? '/' },
    })
  );
});

// Notification click — open or focus the app at the notification URL
self.addEventListener('notificationclick', (event) => {
  event.notification.close();
  event.waitUntil(
    self.clients.matchAll({ type: 'window' }).then((clientList) => {
      const url = event.notification.data?.url ?? '/';
      const existing = clientList.find((c) => c.url.includes(url));
      if (existing) return existing.focus();
      return self.clients.openWindow(url);
    })
  );
});

// Handle subscription change (browser rotated push keys)
self.addEventListener('pushsubscriptionchange', (event: any) => {
  event.waitUntil(
    self.registration.pushManager.subscribe(event.oldSubscription?.options ?? {
      userVisibleOnly: true,
    }).then((subscription) => {
      return fetch('/api/notifications/push-sub', {
        method: 'POST',
        body: JSON.stringify(subscription.toJSON()),
        headers: { 'Content-Type': 'application/json' },
      });
    })
  );
});
```

**1. Push subscription hook** — Create packages/web/src/hooks/usePushSubscription.ts:
```typescript
function urlBase64ToUint8Array(base64String: string): Uint8Array {
  const padding = '='.repeat((4 - base64String.length % 4) % 4);
  const base64 = (base64String + padding).replace(/-/g, '+').replace(/_/g, '/');
  const rawData = window.atob(base64);
  const outputArray = new Uint8Array(rawData.length);
  for (let i = 0; i < rawData.length; ++i) {
    outputArray[i] = rawData.charCodeAt(i);
  }
  return outputArray;
}
```

- `useSubscribeToPush()`: Returns `{ subscribe, unsubscribe, isSubscribed, isSupported }`.
  - `isSupported`: checks `'serviceWorker' in navigator && 'PushManager' in window` AND `import.meta.env.VITE_VAPID_PUBLIC_KEY` is non-empty
  - `subscribe()`: requests Notification.requestPermission(), then calls `registration.pushManager.subscribe({ userVisibleOnly: true, applicationServerKey: urlBase64ToUint8Array(import.meta.env.VITE_VAPID_PUBLIC_KEY) })`, then POSTs subscription.toJSON() to `/api/notifications/push-sub` via the API client from 06-02
  - `unsubscribe()`: calls `subscription.unsubscribe()`, then DELETEs from server
  - `isSubscribed`: check `registration.pushManager.getSubscription()` on mount
  - If VITE_VAPID_PUBLIC_KEY is empty/undefined, isSupported returns false (graceful degradation)
